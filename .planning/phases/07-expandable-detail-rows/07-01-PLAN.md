**F. Add JavaScript** — insert a `<script>` block BEFORE the closing `</body>` tag:

```html
    <script>
    function toggleDetail(icon) {
        var row = icon.closest('tr').nextElementSibling;
        var isHidden = row.style.display === 'none';
        row.style.display = isHidden ? 'table-row' : 'none';
        icon.textContent = isHidden ? '\u25BC' : '\u25B6';
    }
    function toggleAll() {
        var icons = document.querySelectorAll('.toggle-icon');
        var btn = document.querySelector('.expand-all-btn');
        var allOpen = Array.from(icons).every(function(i) { return i.textContent === '\u25BC'; });
        icons.forEach(function(icon) {
            var row = icon.closest('tr').nextElementSibling;
            row.style.display = allOpen ? 'none' : 'table-row';
            icon.textContent = allOpen ? '\u25B6' : '\u25BC';
        });
        btn.textContent = allOpen ? 'Expand All' : 'Collapse All';
    }
    </script>
```

This is ~15 lines of vanilla JS per user decision. No frameworks. No `<details>` element.
