Create `.github/workflows/update-dashboard.yml` with the complete pipeline.

First create the directory: `mkdir -p .github/workflows`

**Full workflow YAML content:**

```yaml
name: Update Dashboard

on:
  schedule:
    - cron: '0 6 * * *'  # Daily at 6 AM UTC
  workflow_dispatch:       # Manual trigger via Actions UI

permissions:
  contents: read
  pages: write
  id-token: write

concurrency:
  group: dashboard-deploy
  cancel-in-progress: true

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    steps:
      - name: Checkout repository
        uses: actions/checkout@v4
        # CRITICAL: Do NOT add submodules: true — that would clone all 49 submodule repos

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.12'

      - name: Install dependencies
        run: pip install -r requirements.txt

      - name: Collect submodule data
        run: python scripts/collector.py
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}

      - name: Render dashboard
        run: python scripts/renderer.py

      - name: Configure Pages
        uses: actions/configure-pages@v5

      - name: Upload Pages artifact
        uses: actions/upload-pages-artifact@v4
        with:
          path: ./site

      - name: Deploy to GitHub Pages
        id: deployment
        uses: actions/deploy-pages@v4
```

**Key decisions reflected:**
- `permissions: contents: read` — only needs read access to the repo
- `permissions: pages: write, id-token: write` — required for Pages deployment
- `concurrency: cancel-in-progress: true` — prevents concurrent deploy races (Pitfall 12)
- `python-version: '3.12'` — matches ubuntu-latest default, ensures fromisoformat() works with Z suffix
- `pip install -r requirements.txt` — installs jinja2 only (requests is pre-installed)
- `GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}` — auto-provided, no PAT needed (CICD-03)
- `environment: name: github-pages` — required for deploy-pages action
- No `submodules: true` in checkout (Pitfall 5)
- No `fetch-depth` needed (we don't use git history, only API)
